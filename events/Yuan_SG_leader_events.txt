# 执行同化并强化人偶
country_event = {
    id = yuan_sg_leader.11
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        exists = event_target:yuan_sg_assimilate_target
    }
    immediate = {
        every_owned_leader = {
            limit = {
                has_leader_flag = yuan_sg_candidate_temp
            }
            remove_leader_flag = yuan_sg_candidate_temp
        }

        if = {
            limit = {
                event_target:yuan_sg_assimilate_target = {
                    leader_class = commander
                }
            }
            if = {
                limit = {
                    event_target:yuan_sg_assimilate_target = {
                        has_base_skill >= 8
                    }
                }
                every_owned_leader = {
                    limit = {
                        has_leader_flag = yuan_sg_leader_node
                        leader_class = commander
                    }
                    add_skill_without_trait_selection = 3
                }
            }
            else_if = {
                limit = {
                    event_target:yuan_sg_assimilate_target = {
                        has_base_skill >= 5
                    }
                }
                every_owned_leader = {
                    limit = {
                        has_leader_flag = yuan_sg_leader_node
                        leader_class = commander
                    }
                    add_skill_without_trait_selection = 2

country_event = {
    id = yuan_sg_leader.3
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_ai = no
    }
    immediate = {
        if = {
            limit = {
                OR = {
                    has_origin = origin_time
                    has_origin = origin_time_end
                    has_origin = origin_time_end_bad
                    has_origin = origin_time_beginning
                    has_origin = origin_time_beginning_bad
                    has_origin = origin_timesingularity
                    has_origin = origin_timesingularity_bad
                    has_origin = origin_endless_time_virtuality
                    has_origin = origin_endless_time_synthetic
                    has_origin = origin_endless_time_cybernetic
                    has_origin = origin_endless_time_genetic
                    has_origin = origin_endless_time_psionic
                }
                ruler = {
                    not = {
                        has_ruler_trait = leader_trait_ruler_time
                    }
                }
            }
            ruler = {
                save_global_event_target_as = origin_time_ruler
                add_trait = {
                    trait = leader_trait_ruler_time
                    show_message = no
                }
            }
        }
        if = {
            limit = {
                OR = {
                    has_authority = auth_dictatorial
                    has_authority = auth_imperial
                    # has_authority = auth_hive_mind
                    # has_authority = auth_machine_intelligence
                }
                ruler = {
                    not = {
                        has_ruler_trait = leader_trait_sg_orthis
                    }
                }
                # owner_main_species = {
                #     OR = {
                #         is_archetype = MACHINE
                #         is_archetype = ROBOT
                #     }
                # }
            }
            ruler = {
                add_trait = {
                    trait = leader_trait_sg_orthis
                    show_message = no
                }
            }
        }
    }
